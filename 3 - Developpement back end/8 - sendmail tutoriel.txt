.env
....
MAIL_MAILER=smtp  #Sample Mail Transfer Protocol
MAIL_HOST=smtp.gmail.com
MAIL_PORT=465
MAIL_USERNAME=ousrah@gmail.com
MAIL_PASSWORD=xxxxxxxxxxxxxxxxxx  #c'est le mot de passe lié à l'application
MAIL_ENCRYPTION=null
MAIL_FROM_ADDRESS=ousrah@gmail.com
MAIL_FROM_NAME="mailIsmo"
....

config.mail.php
....
 'mailers' => [
        'smtp' => [
            'transport' => 'smtp',
            'host' => env('MAIL_HOST', 'smtp.mailgun.org'),
            'port' => env('MAIL_PORT', 587),
            'encryption' => env('MAIL_ENCRYPTION', 'tls'),
            'username' => env('MAIL_USERNAME'),
            'password' => env('MAIL_PASSWORD'),
            'timeout' => null,
            'local_domain' => env('MAIL_EHLO_DOMAIN'),


            'auth_mode'  => null,
            'verify_peer'       => false,

        ],
...
